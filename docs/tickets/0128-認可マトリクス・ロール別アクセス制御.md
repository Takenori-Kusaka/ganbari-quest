### 設計

#### hooks.server.ts でのルート保護

```typescript
// AUTH_MODE=cognito の場合
function authorize(path: string, identity: Identity | null, context: AuthContext | null): AuthResult {
  // 認証不要ルート
  if (path.startsWith('/auth/') || path === '/api/health') {
    return { allowed: true };
  }

  // 未認証 → ログインへ
  if (!identity || !context) {
    return { allowed: false, redirect: '/auth/login', status: 401 };
  }

  // child ロール: /admin/* へのアクセス禁止
  if (context.role === 'child' && path.startsWith('/admin')) {
    return { allowed: false, redirect: `/child/${context.childId}`, status: 403 };
  }

  return { allowed: true };
}
```

#### API レイヤでのテナントスコープ

```typescript
// +server.ts でのテナントスコープ検証
export async function GET({ locals }: RequestEvent) {
  const { context } = locals;
  if (!context?.tenantId) {
    throw error(401, 'Unauthorized');
  }
  // context.tenantId をサービス層に渡す
  const data = await service.getByTenant(context.tenantId);
  return json(data);
}
```

#### child ロールの自分データ制限

```typescript
// child は自分の childId のデータのみアクセス可能
if (context.role === 'child') {
  const requestedChildId = Number(params.childId);
  if (requestedChildId !== context.childId) {
    throw error(403, 'Access denied');
  }
```
